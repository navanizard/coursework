# rank movies by popularity (number of ratings) and compute the cdf, or fraction 
# of all views covered by the top-k movies https://speakerdeck.com/jhofman/modeling-social-data-lecture-2-introduction-to-counting?slide=30
# hint: use dplyr's rank and arrange functions, and the base R sum and cumsum functions
# store the result in a new data frame so you can use it in creating figure 2 from the paper below
ratings_with_rank_and_cdf_attributes <- ratings %>%
  group_by(movie_id) %>%
  summarise(num_of_ratings_per_movie = n()) %>%
  mutate(movie_rank = rank(-num_of_ratings_per_movie, ties.method = 'first')) %>%
  arrange(movie_rank) %>%
  mutate(total_num_of_ratings = sum(num_of_ratings_per_movie), 
         cum_sum_of_ratings = cumsum(num_of_ratings_per_movie))

# plot the CDF of movie popularity
ratings_with_rank_and_cdf_attributes %>%
  ggplot(aes(x = movie_rank, y = cum_sum_of_ratings / total_num_of_ratings)) +
  geom_line() +
  labs(x = 'Movie Rank', 
       y = 'Cumulative Distribution Function (CDF)', 
       title = 'Fraction of All Views Covered by the Top K Movies') + 
  scale_y_continuous(labels = percent_format(accuracy = 1))

4 Anatomy of the long tail

# generate the equivalent of figure 2a of this paper:
# note: don't worry about the "null model" lines
# just do the solid lines and dotted line (optional)
# https://5harad.com/papers/long_tail.pdf
full_tbl <- left_join(ratings, ratings_with_rank_and_cdf_attributes)
## Joining with `by = join_by(movie_id)`
ratings_with_min_ranks_for_satisfactions <- full_tbl %>%
  group_by(user_id) %>%
  summarize(num_of_ratings_per_user = n(),
            min_rank_100 = max(movie_rank),
            min_rank_90 = quantile(movie_rank, 0.9)) %>% 
  filter(num_of_ratings_per_user >= 10) 
  
ratings_with_min_ranks_for_satisfactions %>%
  pivot_longer(names_to = "satisfaction_level", values_to = "num_of_ppl_satisfied_at_this_rank", 3:4) %>%
  group_by(satisfaction_level) %>%
  arrange(num_of_ppl_satisfied_at_this_rank) %>% 
  mutate(total_num_of_users = sum(num_of_ppl_satisfied_at_this_rank), 
         cum_sum_of_ppl_satisfied = cumsum(num_of_ppl_satisfied_at_this_rank)) %>%
  ggplot(aes(x = num_of_ppl_satisfied_at_this_rank, y = cum_sum_of_ppl_satisfied / total_num_of_users, color = satisfaction_level)) +
  geom_line() + 
  labs(x = 'Inventory Size (number of top-k movies offered)',
       y = 'Fraction of Users Satisfied',
       title = 'User Satisfaction Curves for Movies As a Function of Inventory Size',
       color = 'Satisfaction Level') +
  scale_color_discrete(labels = c('100% Satisfaction Level', '90% Satisfaction Level')) +
  scale_y_continuous(labels = percent_format(accuracy = 1))
